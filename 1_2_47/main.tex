\begin{frame}{Problem}
Evaluate:
\begin{align}
\lim_{x \to \frac{\pi}{2}} \frac{\sin^2(2x)}{\left(x - \frac{\pi}{2}\right)^2}
\end{align}
\end{frame}



\begin{frame}{Solution}
\begin{align}
\lim_{x \to \frac{\pi}{2}} &\frac{\sin^2(2x)}{\left(x - \frac{\pi}{2}\right)^2}\\
&\frac{0}{0} \text{ form}
\end{align}

So, we use L'Hopital's Rule :

\begin{align}
\lim_{x \to a} \frac{f(x)}{g(x)} 
= \lim_{x \to a} \frac{f'(x)}{g'(x)}
\end{align}

Differentiate:
\begin{align}
    \frac{d}{dx}[\sin^2(2x)] &= 4\sin(2x)\cos(2x) \\
\frac{d}{dx}[(x - \frac{\pi}{2})^2] &= 2(x - \frac{\pi}{2})
\end{align}

\end{frame}

%------------------------------------------------------------
\begin{frame}{Solution}
\begin{align}
\frac{2\sin(2x)\cos(2x)}{x - \frac{\pi}{2}}
\end{align}

We know that:
\begin{align}
2\sin(2x)\cos(2x) = \sin(4x)
\end{align}

\begin{align}
\Rightarrow \lim_{x \to \frac{\pi}{2}} \frac{\sin(4x)}{x - \frac{\pi}{2}}
\end{align}
\begin{center}
Again  
$\frac{0}{0}$ form
\end{center}
So, differentiate again
\end{frame}

\begin{frame}{Solution}
\begin{align}
    \Rightarrow &\lim_{x \to \frac{\pi}{2}} 4\cos(4x) \\
    \Rightarrow & 4 
    \end{align}

    \begin{align}
\lim_{x \to \frac{\pi}{2}} \frac{\sin^2(2x)}{\left(x - \frac{\pi}{2}\right)^2}
=\boxed{4}
\end{align}
\end{frame}

import matplotlib.pyplot as plt

# Define the function
def f(x):

# Create x values avoiding division by zero

x = x[x != np.pi/2]

# Compute y values
y = f(x)

# Plot the function
plt.figure(figsize=(10, 5))
plt.plot(x, y, label=r"$f(x) = \frac{\sin^2(2x)}{(x - \frac{\pi}{2})^2}$")

\begin{frame}{Problem}

A memoryless source emits $n$ symbols each with a probability $p$. The entropy of the source as a function of $n$ is:

\begin{enumerate}
    \item[(a)] increases as $\log n$
    \item[(b)] decreases as $\log(1/n)$
    \item[(c)] increases as $n$
    \item[(d)] increases as $n \log n$
\end{enumerate}
\end{frame}

\begin{frame}{Solution}
    

Since all $n$ symbols are equally likely,

\begin{align}
p &= \frac{1}{n}
\end{align}

The entropy of a discrete memoryless source is:

\begin{align}
H &= - \sum_{i=1}^{n} p \log_2 p
\end{align}

Substituting $p = \frac{1}{n}$:
\end{frame}
\begin{frame}{Solution}
\begin{align}
H &= - \sum_{i=1}^{n} \frac{1}{n} \log_2 \left(\frac{1}{n}\right) \\
  &= - n \cdot \frac{1}{n} \log_2 \left(\frac{1}{n}\right) \\
  &= - \log_2 \left(\frac{1}{n}\right) \\
  &= \log_2 n
\end{align}

\bigskip

\textbf{Final Answer:}

\begin{align}
H &= \log _2n
\end{align}

\[
\boxed{\text{Option (a): increases as } \log n}
\]
\end{frame}
